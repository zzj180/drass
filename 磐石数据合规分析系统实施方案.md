class ComplianceDemoService:
    async def process_demo_data(
        self, 
        data: Dict[str, Any], 
        compliance_rules: List[str]
    ) -> ComplianceAnalysisResult:
        """
        演示数据合规处理流程
        """
        # 1. 数据分类识别
        data_classification = await self._classify_data(data)
        
        # 2. 合规规则匹配
        applicable_rules = await self._match_compliance_rules(
            data_classification, compliance_rules
        )
        
        # 3. 风险评估
        risk_assessment = await self._assess_compliance_risks(
            data, applicable_rules
        )
        
        # 4. 生成建议
        recommendations = await self._generate_recommendations(
            risk_assessment
        )
        
        return ComplianceAnalysisResult(
            classification=data_classification,
            risks=risk_assessment,
            recommendations=recommendations
        )

# services/main-app/app/services/audit_service.py
class AuditService:
    def __init__(self):
        self.event_handlers = {
            'document_upload': self._log_document_upload,
            'document_access': self._log_document_access,
            'compliance_analysis': self._log_compliance_analysis,
            'user_login': self._log_user_login,
            'data_export': self._log_data_export,
            'permission_change': self._log_permission_change
        }
    
    async def log_audit_event(
        self, 
        event_type: str, 
        user_id: str, 
        details: Dict[str, Any],
        request_id: str = None
    ) -> AuditLogEntry:
        """
        记录审计事件
        """
        timestamp = datetime.utcnow()
        
        # 获取用户信息
        user_info = await self._get_user_info(user_id)
        
        # 获取请求上下文
        request_context = await self._get_request_context(request_id)
        
        # 创建审计日志条目
        audit_entry = AuditLogEntry(
            id=str(uuid4()),
            timestamp=timestamp,
            event_type=event_type,
            user_id=user_id,
            user_name=user_info.get('name', 'Unknown'),
            user_role=user_info.get('role', 'Unknown'),
            details=details,
            request_id=request_id,
            ip_address=request_context.get('ip_address'),
            user_agent=request_context.get('user_agent'),
            success=details.get('success', True),
            error_message=details.get('error_message')
        )
        
        # 存储到数据库
        await self._store_audit_log(audit_entry)
        
        # 实时推送到前端
        await self._broadcast_audit_event(audit_entry)
        
        return audit_entry
```

# services/main-app/app/services/monitoring_service.py
class ComplianceMonitoringService:
    def __init__(self):
        self.risk_thresholds = {
            'data_access_frequency': 100,  # 每小时访问次数
            'sensitive_data_exposure': 0,   # 敏感数据暴露次数
            'compliance_violations': 0      # 合规违规次数
        }
    
    async def monitor_compliance_metrics(self) -> MonitoringResult:
        """
        实时监控合规指标
        """
        current_time = datetime.utcnow()
        
        # 获取各项指标
        metrics = {
            'data_access_count': await self._get_data_access_count(current_time),
            'sensitive_data_queries': await self._get_sensitive_data_queries(current_time),
            'compliance_score': await self._calculate_compliance_score(),
            'risk_incidents': await self._get_risk_incidents(current_time)
        }
        
        # 检查阈值和生成告警
        alerts = await self._check_thresholds_and_generate_alerts(metrics)
        
        return MonitoringResult(
            timestamp=current_time,
            metrics=metrics,

# services/main-app/app/websocket/audit_websocket.py
class AuditWebSocketManager:
    def __init__(self):
        self.active_connections: List[WebSocket] = []
        self.user_connections: Dict[str, List[WebSocket]] = {}
    
    async def connect(self, websocket: WebSocket, user_id: str):
        await websocket.accept()
        self.active_connections.append(websocket)
        
        if user_id not in self.user_connections:
            self.user_connections[user_id] = []
        self.user_connections[user_id].append(websocket)
    
    async def broadcast_audit_event(self, audit_entry: AuditLogEntry):
        """
        广播审计事件到所有连接的客户端
        """
        message = {
            'type': 'audit_event',
            'data': audit_entry.dict()
        }
        
        disconnected = []
        for connection in self.active_connections:
            try:
                await connection.send_json(message)
            except ConnectionClosedOK:
                disconnected.append(connection)
        
        # 清理断开的连接
        for connection in disconnected:
            self.active_connections.remove(connection)
```
