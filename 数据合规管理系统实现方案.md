class ComplianceMonitoringService:
    def __init__(self):
        self.risk_thresholds = {
            'data_access_frequency': 100,  # 每小时访问次数
            'sensitive_data_exposure': 0,   # 敏感数据暴露次数
            'compliance_violations': 0      # 合规违规次数
        }
    
    async def monitor_compliance_metrics(self) -> MonitoringResult:
        """
        实时监控合规指标
        """
        current_time = datetime.utcnow()
        
        # 获取各项指标
        metrics = {
            'data_access_count': await self._get_data_access_count(current_time),
            'sensitive_data_queries': await self._get_sensitive_data_queries(current_time),
            'compliance_score': await self._calculate_compliance_score(),
            'risk_incidents': await self._get_risk_incidents(current_time)
        }
        
        # 检查阈值和生成告警
        alerts = await self._check_thresholds_and_generate_alerts(metrics)
        
        return MonitoringResult(
            timestamp=current_time,
            metrics=metrics,
            alerts=alerts
        )
    
    async def _check_thresholds_and_generate_alerts(
        self, 
        metrics: Dict[str, Any]
    ) -> List[Alert]:
        """
        检查阈值并生成告警
        """
        alerts = []
        
        if metrics['data_access_count'] > self.risk_thresholds['data_access_frequency']:
            alerts.append(Alert(
                level='warning',
                type='high_data_access',
                message=f"数据访问频率异常：{metrics['data_access_count']}次/小时",
                timestamp=datetime.utcnow(),
                recommendations=["检查访问来源", "验证访问权限", "启用访问限制"]
            ))
        
        return alerts

class DocumentPreprocessor:
    def __init__(self):
        self.supported_formats = {
            '.pdf': self._process_pdf,
            '.docx': self._process_docx,
            '.txt': self._process_txt,
            '.xlsx': self._process_excel
        }
    
    async def preprocess_document(
        self, 
        file_content: bytes, 
        filename: str
    ) -> PreprocessResult:
        """
        文档预处理：提取文本、生成预览、合规检查
        """
        file_ext = Path(filename).suffix.lower()
        
        if file_ext not in self.supported_formats:
            raise ValueError(f"不支持的文件格式: {file_ext}")
        
        # 1. 提取文本内容
        text_content = await self.supported_formats[file_ext](file_content)
        
        # 2. 生成文档摘要
        summary = await self._generate_summary(text_content)
        
        # 3. 敏感信息检测
        sensitive_info = await self._detect_sensitive_information(text_content)
        
        # 4. 合规预检查
        compliance_check = await self._preliminary_compliance_check(
            text_content, sensitive_info
        )
        
        return PreprocessResult(
            text_content=text_content,
            summary=summary,
            sensitive_info=sensitive_info,
            compliance_status=compliance_check,
            metadata={
                'word_count': len(text_content.split()),
                'char_count': len(text_content),
                'language': self._detect_language(text_content)
            }
        )
    
    async def _detect_sensitive_information(self, text: str) -> List[SensitiveInfo]:
        """
        检测敏感信息
        """
        patterns = {
            'id_card': r'\d{17}[\dXx]',  # 身份证号
            'phone': r'1[3-9]\d{9}',    # 手机号
            'email': r'[\w\.-]+@[\w\.-]+\.\w+',  # 邮箱
            'bank_card': r'\d{16,19}',   # 银行卡号
        }
        
        sensitive_items = []
        for info_type, pattern in patterns.items():
            matches = re.findall(pattern, text)
            for match in matches:
                sensitive_items.append(SensitiveInfo(
                    type=info_type,
                    value=self._mask_sensitive_value(match),
                    position=text.find(match),
                    confidence=0.9
                ))
        
        return sensitive_items

# services/main-app/app/websocket/audit_websocket.py
class AuditWebSocketManager:
    def __init__(self):
        self.active_connections: List[WebSocket] = []
        self.user_connections: Dict[str, List[WebSocket]] = {}
    
    async def connect(self, websocket: WebSocket, user_id: str):
        await websocket.accept()
        self.active_connections.append(websocket)
        
        if user_id not in self.user_connections:
            self.user_connections[user_id] = []
        self.user_connections[user_id].append(websocket)
    
    def disconnect(self, websocket: WebSocket, user_id: str):
        self.active_connections.remove(websocket)
        if user_id in self.user_connections:
            self.user_connections[user_id].remove(websocket)
    
    async def broadcast_audit_event(self, audit_entry: AuditLogEntry):
        """
        广播审计事件到所有连接的客户端
        """
        message = {
            'type': 'audit_event',
            'data': audit_entry.dict()
        }
        
        disconnected = []
        for connection in self.active_connections:
            try:
                await connection.send_json(message)
            except ConnectionClosedOK:
                disconnected.append(connection)
        
        # 清理断开的连接
        for connection in disconnected:
            self.active_connections.remove(connection)

# WebSocket端点
@app.websocket("/ws/audit/{user_id}")
async def audit_websocket_endpoint(websocket: WebSocket, user_id: str):
    await audit_ws_manager.connect(websocket, user_id)
    try:
        while True:
            # 保持连接活跃
            await websocket.receive_text()
    except WebSocketDisconnect:
        audit_ws_manager.disconnect(websocket, user_id)
