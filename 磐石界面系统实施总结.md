## 🎨 使用示例

### 1. 数据卡片使用
```tsx
import { DataCard } from '@/components/BedrockUI';

<DataCard
  title="合规评分"
  value="87%"
  subtitle="本月平均合规率"
  status="compliant"
  trend="up"
  trendValue="+12%"
  progress={87}
  icon={<CheckCircleIcon />}
  onClick={() => handleCardClick()}
/>
```

### 2. 状态徽章使用
```tsx
import { StatusBadge } from '@/components/BedrockUI';

<StatusBadge status="compliant" showIcon pulse />
<StatusBadge status="risk" showIcon glow />
```

### 3. 渐变按钮使用
```tsx
import { GradientButton } from '@/components/BedrockUI';

<GradientButton
  variant="primary"
  glow
  startIcon={<SecurityIcon />}
  onClick={handleClick}
>
  开始合规检查
</GradientButton>
```
